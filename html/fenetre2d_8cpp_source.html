334 void fenetre2D::afficherDernierEtat() //change par rapport à la fenetre 1D
335 {
336  if (m_simulateur!= nullptr)
337  {
338  Etat const& etat = m_simulateur->Dernier();

340  {
341  const Cell& cellule = *it;
342  switch(cellule.GetEtat())
343  {
344  case BLANC:
345  m_grille->item(cellule.GetX(),cellule.GetY())->setText("0");
346  m_grille->item(cellule.GetX(),cellule.GetY())->setBackgroundColor("white");
347  m_grille->item(cellule.GetX(),cellule.GetY())->setTextColor("white");
348  break;
349  case NOIR:
350  m_grille->item(cellule.GetX(),cellule.GetY())->setText("1");
351  m_grille->item(cellule.GetX(),cellule.GetY())->setBackgroundColor("black");
352  m_grille->item(cellule.GetX(),cellule.GetY())->setTextColor("black");;
353  break;
354  case VERT:
355  m_grille->item(cellule.GetX(),cellule.GetY())->setText("2");
356  m_grille->item(cellule.GetX(),cellule.GetY())->setBackgroundColor("green");
357  m_grille->item(cellule.GetX(),cellule.GetY())->setTextColor("green");
358  break;
359  case ROUGE:
360  m_grille->item(cellule.GetX(),cellule.GetY())->setText("3");
361  m_grille->item(cellule.GetX(),cellule.GetY())->setBackgroundColor("red");
362  m_grille->item(cellule.GetX(),cellule.GetY())->setTextColor("red");
363  break;
364  default:
365  std::stringstream flux;
366  flux << cellule.GetEtat();
367  m_grille->item(0,cellule.GetY())->setText(flux.str().c_str());
368  m_grille->item(0,cellule.GetY())->setBackgroundColor("white");
